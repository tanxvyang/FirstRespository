create table t_dept(
	id int primary key auto_increment,
	name varchar(10)
);

create table t_emp(
	id int primary key auto_increment,
	name varchar(10),
	salary double,
	dept_id int references t_dept(id),
	manager_id int references t_emp(id)
);
